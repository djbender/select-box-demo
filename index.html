  optgroups: [
    {value: 'usa', label: 'United States'},

// Basic Select
$('#selectize-basic').selectize({
  create: false,
  sortField: 'text',
  searchField: 'text',
  options: countries,
  placeholder: 'Select a country...',
  maxItems: 1
});

// Multi-Select with Tags
$('#selectize-tags').selectize({
  delimiter: ',',
  persist: false,
  create: function(input) {
    return {
      value: input.toLowerCase().replace(/\s+/g, '-'),
      text: input
    }
  },
  options: programmingTags,
  placeholder: 'Choose tags...',
  maxItems: null,
  closeAfterSelect: true,
  plugins: ['remove_button']
});

// Option Groups
$('#selectize-groups').selectize({
  sortField: 'text',
  searchField: 'text',
  optgroups: cityGroups.optgroups,
  options: cityGroups.options,
  placeholder: 'Select a city...',
  optgroupField: 'optgroup',
  labelField: 'text',
  valueField: 'value'
});

  optgroupField: 'optgroup',
  labelField: 'text',
  valueField: 'value',
  plugins: ['clear_button']
});

Key Code Differences

Initialization

Selectize: Uses jQuery plugin syntax with $('#element').selectize()

Tom Select: Uses constructor syntax with new TomSelect('#element')

Dependencies

Selectize: Requires jQuery as a dependency

Tom Select: No dependencies - works with vanilla JavaScript

Plugin System

Selectize: Has essential plugins like remove_button and clear_button

Tom Select: Larger plugin ecosystem with additional options like virtual scrolling and checkbox modes

Configuration

Selectize: Simpler configuration options

Tom Select: More granular control with options like allowEmptyOption

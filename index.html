Implementation Code

// Basic Select
$('#selectize-basic').selectize({
  create: false,
  sortField: 'text',
  searchField: 'text',
  options: [
    {value: 'us', text: 'United States'},
    {value: 'uk', text: 'United Kingdom'},
    {value: 'ca', text: 'Canada'}
  ]
});

// Multi-Select with Tags
$('#selectize-tags').selectize({
  delimiter: ',',
  persist: false,
  create: function(input) {
    return {
      value: input,
      text: input
    }
  },
  options: [
    {value: 'javascript', text: 'JavaScript'},
    {value: 'python', text: 'Python'},
    {value: 'react', text: 'React'}
  ]
});

// Option Groups
$('#selectize-groups').selectize({
  optgroups: [
    {value: 'usa', label: 'United States'},
    {value: 'canada', label: 'Canada'}
  ],
  options: [
    {value: 'nyc', text: 'New York', optgroup: 'usa'},
    {value: 'sf', text: 'San Francisco', optgroup: 'usa'},
    {value: 'toronto', text: 'Toronto', optgroup: 'canada'}
  ]
});

// Basic Select
new TomSelect('#tom-basic', {
  create: false,
  sortField: 'text',
  searchField: 'text',
  options: [
    {value: 'us', text: 'United States'},
    {value: 'uk', text: 'United Kingdom'},
    {value: 'ca', text: 'Canada'}
  ]
});

// Multi-Select with Tags
new TomSelect('#tom-tags', {
  delimiter: ',',
  persist: false,
  create: function(input) {
    return {
      value: input,
      text: input
    }
  },
  options: [
    {value: 'javascript', text: 'JavaScript'},
    {value: 'python', text: 'Python'},
    {value: 'react', text: 'React'}
  ],
  plugins: ['remove_button']
});

// Option Groups
new TomSelect('#tom-groups', {
  optgroups: [
    {value: 'usa', label: 'United States'},
    {value: 'canada', label: 'Canada'}
  ],
  options: [
    {value: 'nyc', text: 'New York', optgroup: 'usa'},
    {value: 'sf', text: 'San Francisco', optgroup: 'usa'},
    {value: 'toronto', text: 'Toronto', optgroup: 'canada'}
  ]
});
